95      /**
96       * Copies over a file from the sourceDirectory to the targetDirectory preserving its relative subdirectories.
97       *
98       * @param sourceDirectory where the main source directory is
99       * @param targetDirectory where the target directory is
100      * @param sourceFile which file to copy to the target directory
101      * @param forceOverwrite if timestamps should be ignored
102      * @throws IOException if something goes wrong while copying
103      */
104     public static void copyToFolder(File sourceDirectory, File targetDirectory, File sourceFile, boolean forceOverwrite)
105         throws IOException {
106 
107         final String relativeFile = PathUtils.toRelative(
108             sourceDirectory.getAbsoluteFile(),
109             sourceFile.getAbsolutePath()
110         );
111         final File targetFile = new File(targetDirectory, relativeFile);
112 
113         if (forceOverwrite) {
114             FileUtils.copyFile(sourceFile, targetFile);
115         } else {
116             FileUtils.copyFileIfModified(sourceFile, targetFile);
117         }
118     }

120     /**
121      * Unzips all files to the given directory (using jar).
122      *
123      * @param log Logging
124      * @param targetDirectory where to unpack the files to
125      * @param elements list of files to unpack
126      * @param exec execution helper used to unphar elements
127      * @throws IOException if something goes wrong while copying
128      */
129     public static void unzipElements(Log log, File targetDirectory, List<String> elements, IPhpExecution exec)
130         throws IOException {
131         Preconditions.checkArgument(
132             !targetDirectory.exists() || targetDirectory.isDirectory(),
133             "Destination Directory");
134 
135         targetDirectory.mkdirs();
136         if (!targetDirectory.exists()) {
137             throw new IllegalStateException("Could not create target directory " + targetDirectory.getAbsolutePath());
138         }
139         
140         log.debug(elements.toString());
141 
142         for (String element : elements) {
143             log.debug("unpacking " + element);
144             final File sourceFile = new File(element);
145             if (sourceFile.isFile()) {
146                 final int pos = sourceFile.getName().lastIndexOf('.');
147                 String extension = sourceFile.getName();
148                 if (pos != -1) {
149                     extension = extension.substring(pos + 1);
150                 }
151                 
152                 if ("jar".equals(extension)) {
153                     // for backward compatibility to phpmaven1; there we build jar instead of phar
154                     unjar(log, sourceFile, targetDirectory);
155                 } else if ("phar".equals(extension)) {
156                     unphar(log, targetDirectory, exec, sourceFile);
157                 } else if ("zip".equals(extension)) {
158                     // although jar and zips are compatible to each other this is a implementation detail of jvm.
159                     // we should not depend on it. so let us divide it.
160                     unzip(log, sourceFile, targetDirectory);
161                 } else {
162                     throw new IOException("Unknown archive format. Unable to extract " + sourceFile.getAbsolutePath());
163                 }
164             }
165         }
166     }

168     /**
169      * Unphar given file to destination directory.
170      * 
171      * @param log Logging
172      * @param targetDirectory where to unpack the files to
173      * @param exec execution helper used to unphar elements
174      * @param sourceFile the jar source file
175      * @throws IOException if something goes wrong while copying
176      */
177     public static void unphar(Log log, File targetDirectory, IPhpExecution exec,
178             final File sourceFile) throws IOException {
179         log.debug("unphar " + sourceFile.getAbsolutePath());
180         final String pharFilenameMasked = sourceFile.getAbsolutePath().replace("\\", "\\\\");
181         final String targetMasked = targetDirectory.getAbsolutePath().replace("\\", "\\\\");
182         final String contents =
183             UNPHAR_SNIPPET.
184             replace("$:{pharFile}", pharFilenameMasked).
185                 replace("$:{targetDir}", targetMasked);
186         try {
187             exec.executeCode(null, contents);
188         } catch (PhpException e) {
189             throw new IOException(
190                     "Error while execution unphar script. Unable to extract "
191                     + sourceFile.getAbsolutePath(), e);
192         }
193     }

195     /**
196      * Unpacks a jar file.
197      *
198      * @param log Logging
199      * @param jarFile the jar file
200      * @param destDir the destination directory
201      * @throws IOException if something goes wrong
202      */
203     public static void unjar(Log log, File jarFile, File destDir) throws IOException {
204         Preconditions.checkNotNull(jarFile, "JarFile");
205 
206         final JarFile jar = new JarFile(jarFile);
207         log.debug("unjar " + jarFile.getAbsolutePath());
208 
209         final Enumeration<JarEntry> items = jar.entries();
210         while (items.hasMoreElements()) {
211             final JarEntry entry = items.nextElement();
212             unpackJarEntry(entry, jar.getInputStream(entry), destDir);
213         }
214     }

225         Preconditions.checkNotNull(zipFile, "ZipFile");

228         log.debug("unzip " + zipFile.getAbsolutePath());

250     /**
251      * Unpacks a jar stream.
252      *
253      * @param inputStream the jar stream
254      * @param destDir the destination directory
255      * @throws IOException if something goes wrong
256      */
257     public static void unjar(InputStream inputStream, File destDir) throws IOException {
258         Preconditions.checkNotNull(inputStream, "InputStream");
259 
260         final JarInputStream jarInputStream = new JarInputStream(inputStream);
261         while (true) {
262             final JarEntry entry = jarInputStream.getNextJarEntry();
263             if (entry == null) {
264                 break;
265             }
266             unpackJarEntry(entry, jarInputStream, destDir);
267         }
268     }

289     /**
290      * Unpacks a single zip entry.
291      *
292      * @param zipEntry the zip entry
293      * @param zipEntryInputStream the source stream of the entry
294      * @param destDir the destination directory
295      * @throws IOException if something goes wrong
296      */
297     public static void unpackZipEntry(ZipEntry zipEntry, InputStream zipEntryInputStream, File destDir)
298         throws IOException {
299 
300         Preconditions.checkNotNull(zipEntry, "ZipEntry");
301         Preconditions.checkNotNull(zipEntryInputStream, "ZipEntryInputStream");
302         Preconditions.checkNotNull(destDir, "Destination Directory");
303         Preconditions.checkArgument(!destDir.exists() || destDir.isDirectory(), "Destination Directory");
304 
305         // final name
306         final File destFile = new File(destDir, zipEntry.getName());
307 
308         // already there
309         if (destFile.exists()) {
310             return;
311         }
312 
313         // just a directory to create
314         if (zipEntry.isDirectory()) {
315             destFile.mkdir();
316             return;
317         }
318 
319         OutputStream out = null;
320         try {
321             out = new FileOutputStream(destFile);
322             IOUtil.copy(zipEntryInputStream, out);
323         } finally {
324             if (out != null) out.close();
325         }
326     }

328     /**
329      * Reads an url to string; should only be used for non-blocking connections
330      * (f.e. jar file contents and other things).
331      * 
332      * @param url the url to be read
333      * @return the results
334      * @throws IOException thrown on problems while reading.
335      */
336     public static String readUrl(final URL url) throws IOException {
337         final InputStream stream = url.openStream();
338         final BufferedReader in = new BufferedReader(new InputStreamReader(stream));
339         String inputLine;
340         final StringBuffer result = new StringBuffer();
341         while ((inputLine = in.readLine()) != null) {
342             if (result.length() > 0) {
343                 result.append("\n");
344             }
345             result.append(inputLine);
346         }
347         in.close();
348         return result.toString();
349     }
